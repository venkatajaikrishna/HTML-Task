        </div>

        <!-- Task 2: Check if character is vowel or consonant -->
        <div class="form-group">
            <label for="charInput">Enter a character to check if it is a vowel or consonant:</label>
            <input type="text" id="charInput" maxlength="1" class="form-control">
            <button onclick="checkVowelOrConsonant()" class="btn btn-primary mt-2">Check</button>
            <div id="vowelConsonantResult" class="mt-2"></div>
        </div>

        <!-- Task 3: Count of even and odd numbers from 1 to 999 -->
        <div class="form-group">
            <button onclick="countEvenAndOdd()" class="btn btn-primary mt-2">Count Even and Odd from 1 to 999</button>
            <div id="evenOddCountResult" class="mt-2"></div>
        </div>

        <!-- Task 4: Count occurrence of a particular character in a string -->
        <div class="form-group">
            <label for="stringInput">Enter a string:</label>
            <input type="text" id="stringInput" class="form-control">
            <label for="charCountInput" class="mt-2">Enter the character to count:</label>
            <input type="text" id="charCountInput" maxlength="1" class="form-control">
            <button onclick="countCharacterOccurrences()" class="btn btn-primary mt-2">Count</button>
            <div id="charCountResult" class="mt-2"></div>
        </div>

        <!-- Task 5: Sum and average of array elements -->

        <!-- Task 6: Largest number in an array -->
        <div class="form-group">
            <button onclick="findLargestNumber()" class="btn btn-primary mt-2">Find Largest Number</button>
            <div id="largestNumberResult" class="mt-2"></div>

    <script src="https://code.jquery.com/jquery-3.5.1.min.js"></script>

    <script>
        // Task 1: Display even numbers up to n
        function displayEvenNumbers() {
            const n = parseInt(document.getElementById('evenNumberInput').value);
            let result = '';
            for (let i = 2; i <= n; i += 2) {
                result += i + ' ';
            }
            document.getElementById('evenNumberResult').textContent = result.trim();
        }

        // Task 2: Check if character is vowel or consonant
        function checkVowelOrConsonant() {
            const char = document.getElementById('charInput').value.toLowerCase();
            const vowels = ['a', 'e', 'i', 'o', 'u'];
            let result = '';
            if (vowels.includes(char)) {
                result = `${char} is a vowel.`;
            } else {
                result = `${char} is a consonant.`;
            }
            document.getElementById('vowelConsonantResult').textContent = result;
        }

        // Task 3: Count of even and odd numbers from 1 to 999
        function countEvenAndOdd() {
            let evenCount = 0;
            let oddCount = 0;
            for (let i = 1; i <= 999; i++) {
                if (i % 2 === 0) {
                    evenCount++;
                } else {
                    oddCount++;
                }
            }
            document.getElementById('evenOddCountResult').textContent = `Even: ${evenCount}, Odd: ${oddCount}`;
        }

        // Task 4: Count occurrence of a particular character in a string
        function countCharacterOccurrences() {
            const str = document.getElementById('stringInput').value;
            const char = document.getElementById('charCountInput').value;
            let count = 0;
            for (let i = 0; i < str.length; i++) {
                if (str[i] === char) {
                    count++;
                }
            }
            document.getElementById('charCountResult').textContent = `Count of '${char}' in "${str}": ${count}`;
        }

        // Task 5: Sum and average of array elements
        function calculateSumAndAverage() {
            const array = [1, 9, 8];
            let sum = 0;
            for (let i = 0; i < array.length; i++) {
                sum += array[i];
            }
            const average = sum / array.length;
            document.getElementById('sumAverageResult').textContent = `Sum: ${sum}, Average: ${average.toFixed(2)}`;
        }

        // Task 6: Largest number in an array
        function findLargestNumber() {
            const array = [1, 9, 8];
            let largest = array[0];
            for (let i = 1; i < array.length; i++) {
                if (array[i] > largest) {
                    largest = array[i];
                }
            }
            document.getElementById('largestNumberResult').textContent = `Largest number in the array: ${largest}`;
        }
    </script>
